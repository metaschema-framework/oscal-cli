6   package gov.nist.secauto.oscal.tools.cli.core.commands;
7   
8   import gov.nist.secauto.metaschema.cli.commands.AbstractValidateContentCommand;
9   import gov.nist.secauto.metaschema.cli.processor.CLIProcessor.CallingContext;
10  import gov.nist.secauto.metaschema.cli.processor.command.CommandExecutionException;
11  import gov.nist.secauto.metaschema.cli.processor.command.ICommandExecutor;
12  import gov.nist.secauto.metaschema.core.model.IModule;
13  import gov.nist.secauto.metaschema.core.model.constraint.IConstraintSet;
14  import gov.nist.secauto.metaschema.core.model.validation.JsonSchemaContentValidator;
15  import gov.nist.secauto.metaschema.core.model.validation.XmlSchemaContentValidator;
16  import gov.nist.secauto.metaschema.databind.IBindingContext;
17  import gov.nist.secauto.metaschema.databind.IBindingContext.ISchemaValidationProvider;
18  import gov.nist.secauto.oscal.lib.OscalBindingContext;
19  import gov.nist.secauto.oscal.lib.model.OscalCompleteModule;
20  
21  import org.apache.commons.cli.CommandLine;
22  import org.json.JSONObject;
23  import org.xml.sax.SAXException;
24  
25  import java.io.IOException;
26  import java.net.URL;
27  import java.util.Set;
28  
29  import edu.umd.cs.findbugs.annotations.NonNull;
30  
31  public abstract class AbstractOscalValidationCommand
32      extends AbstractValidateContentCommand {
33  
34    @NonNull
35    protected abstract XmlSchemaContentValidator getOscalXmlSchemas() throws IOException, SAXException;
36  
37    @NonNull
38    protected abstract JsonSchemaContentValidator getOscalJsonSchema() throws IOException;
39  
40    @Override
41    public ICommandExecutor newExecutor(CallingContext callingContext, CommandLine commandLine) {
42      return new AbstractOscalValidationCommand.OscalValidationCommandExecutor(callingContext, commandLine);
43    }
44  
45    protected class OscalValidationCommandExecutor
46        extends AbstractValidateContentCommand.AbstractValidationCommandExecutor
47        implements ISchemaValidationProvider {
48  
49      protected OscalValidationCommandExecutor(
50          @NonNull CallingContext callingContext,
51          @NonNull CommandLine commandLine) {
52        super(callingContext, commandLine);
53      }
54  
55      @Override
56      protected IBindingContext getBindingContext(@NonNull Set<IConstraintSet> constraintSets) {
57        return OscalBindingContext.builder()
58            .constraintSet(constraintSets)
59            .build();
60      }
61  
62      @Override
63      @NonNull
64      public XmlSchemaContentValidator getXmlSchemas(URL targetResource, IBindingContext bindingContext)
65          throws IOException, SAXException {
66        return getOscalXmlSchemas();
67      }
68  
69      @Override
70      @NonNull
71      public JsonSchemaContentValidator getJsonSchema(JSONObject json, IBindingContext bindingContext)
72          throws IOException {
73        return getOscalJsonSchema();
74      }
75  
76      @Override
77      @NonNull
78      protected ISchemaValidationProvider getSchemaValidationProvider(
79          @NonNull IModule module,
80          @NonNull CommandLine commandLine,
81          @NonNull IBindingContext bindingContext) {
82        return this;
83      }
84  
85      @Override
86      protected IModule getModule(CommandLine commandLine, IBindingContext bindingContext)
87          throws CommandExecutionException {
88        return bindingContext.registerModule(OscalCompleteModule.class);
89      }
90    }
91  }
