109   /**
110    * Execute the list allowed values command.
111    *
112    * @param callingContext
113    *          information about the calling context
114    * @param cmdLine
115    *          the parsed command line details
116    * @throws CommandExecutionException
117    *           if an error occurred while executing the command
118    */
119   @SuppressWarnings({
120       "PMD.OnlyOneReturn", // readability
121       "PMD.AvoidCatchingGenericException",
122       "PMD.CognitiveComplexity",
123       "PMD.CyclomaticComplexity"
124   })
125   protected void executeCommand(
126       @NonNull CallingContext callingContext,
127       @NonNull CommandLine cmdLine) throws CommandExecutionException {
128 
129     List<String> extraArgs = cmdLine.getArgList();
130 
131     Path destination = null;
132     if (!extraArgs.isEmpty()) {
133       destination = MetaschemaCommands.handleDestination(ObjectUtils.requireNonNull(extraArgs.get(0)), cmdLine);
134     }
135 
136     URI currentWorkingDirectory = ObjectUtils.notNull(getCurrentWorkingDirectory().toUri());
137     Set<IConstraintSet> constraintSets = MetaschemaCommands.loadConstraintSets(
138         cmdLine,
139         CONSTRAINTS_OPTION,
140         currentWorkingDirectory);
141 
142     IBindingContext bindingContext;
143     try {
144       bindingContext = OscalBindingContext.builder()
145           .constraintSet(constraintSets)
146           .build();
147     } catch (RuntimeException ex) {
148       throw new CommandExecutionException(ExitCode.RUNTIME_ERROR,
149           String.format("Unable to initialize the binding context. %s", ex.getLocalizedMessage()),
150           ex);
151     }
152 
153     IBoundModule module = bindingContext.registerModule(OscalCompleteModule.class);
154 
155     try {
156       if (destination == null) {
157         Writer stringWriter = new StringWriter();
158         try (PrintWriter writer = new PrintWriter(stringWriter)) {
159           generateAllowedValuesList(module, writer, null);
160         }
161 
162         // Print the result
163         if (LOGGER.isInfoEnabled()) {
164           LOGGER.info(stringWriter.toString());
165         }
166       } else {
167         try (Writer writer = Files.newBufferedWriter(
168             destination,
169             StandardCharsets.UTF_8,
170             StandardOpenOption.CREATE,
171             StandardOpenOption.WRITE,
172             StandardOpenOption.TRUNCATE_EXISTING)) {
173           try (PrintWriter printWriter = new PrintWriter(writer)) {
174             generateAllowedValuesList(module, printWriter, null);
175           }
176         }
177       }
178     } catch (IOException ex) {
179       throw new CommandExecutionException(ExitCode.IO_ERROR, ex);
180     } catch (RuntimeException ex) {
181       throw new CommandExecutionException(ExitCode.RUNTIME_ERROR, ex);
182     }
183   }

185   private static void generateAllowedValuesList(
186       @NonNull IModule module,
187       @NonNull PrintWriter writer,
188       @SuppressWarnings("unused") @Nullable String metapath) throws IOException {
189     AllowedValueCollectingNodeItemVisitor walker = new AllowedValueCollectingNodeItemVisitor();
190 
191     StaticContext staticContext = StaticContext.builder()
192         .defaultModelNamespace(module.getXmlNamespace())
193         .build();
194 
195     IModuleNodeItem moduleNodeItem = INodeItemFactory.instance().newModuleNodeItem(module);
196 
197     // if (metapath != null) {
198     // MetapathExpression filter = MetapathExpression.compile(metapath,
199     // staticContext);
200     //
201     // DynamicContext dynamicContext = new DynamicContext(staticContext);
202     // ISequence<?> sequence = filter.evaluate(moduleNodeItem, dynamicContext);
203     // assert sequence != null;
204     // }
205 
206     DynamicContext dynamicContext = new DynamicContext(staticContext);
207     dynamicContext.disablePredicateEvaluation();
208 
209     walker.visit(moduleNodeItem, dynamicContext);
210 
211     Map<IDefinitionNodeItem<?, ?>,
212         List<AllowedValueCollectingNodeItemVisitor.AllowedValuesRecord>> allowedValuesByTarget
213             = ObjectUtils.notNull(walker.getAllowedValueLocations().stream()
214                 .flatMap(location -> location.getAllowedValues().stream())
215                 .collect(Collectors.groupingBy(AllowedValuesRecord::getTarget,
216                     () -> new TreeMap<>(Comparator.comparing(IDefinitionNodeItem::getMetapath)),
217                     Collectors.mapping(Function.identity(), Collectors.toUnmodifiableList()))));
218 
219     generateYaml(allowedValuesByTarget, writer);
220   }

222   private static void generateYaml(
223       @NonNull Map<IDefinitionNodeItem<?, ?>, List<AllowedValuesRecord>> allowedValuesByTarget,
224       @NonNull PrintWriter writer) throws IOException {
225 
226     YAMLFactoryBuilder builder = YAMLFactory.builder();
227     YAMLFactory factory = ObjectUtils.notNull(builder
228         .enable(YAMLGenerator.Feature.MINIMIZE_QUOTES)
229         .enable(YAMLGenerator.Feature.WRITE_DOC_START_MARKER)
230         .enable(YAMLGenerator.Feature.ALWAYS_QUOTE_NUMBERS_AS_STRINGS)
231         .disable(YAMLGenerator.Feature.SPLIT_LINES)
232         .build());
233 
234     try (YAMLGenerator generator = factory.createGenerator(writer)) {
235 
236       generator.writeStartObject(); // toplevel
237 
238       writeLocations(allowedValuesByTarget, generator);
239 
240       generator.writeEndObject(); // toplevel
241     }
242   }
243 
244   private static void writeLocations(
245       @NonNull Map<IDefinitionNodeItem<?, ?>, List<AllowedValuesRecord>> allowedValuesByTarget,
246       @NonNull YAMLGenerator generator) throws IOException {
247     generator.writeFieldName("locations");
248     generator.writeStartObject(); // locations
249 
250     for (Map.Entry<IDefinitionNodeItem<?, ?>, List<AllowedValuesRecord>> entry : allowedValuesByTarget.entrySet()) {
251       assert entry != null;
252       writeLocation(entry, generator);
253     }
254     generator.writeEndObject(); // locations
255   }
256 
257   private static void writeLocation(
258       @NonNull Map.Entry<IDefinitionNodeItem<?, ?>, List<AllowedValuesRecord>> entry,
259       @NonNull YAMLGenerator generator) throws IOException {
260 
261     IDefinitionNodeItem<?, ?> target = ObjectUtils.notNull(entry.getKey());
262 
263     generator.writeFieldName(metapath(target));
264 
265     generator.writeStartObject(); // metapath
266 
267     writeLocationConstraints(entry, generator);
268 
269     generator.writeEndObject(); // metapath
270   }
271 
272   private static void writeLocationConstraints(
273       @NonNull Entry<IDefinitionNodeItem<?, ?>, List<AllowedValuesRecord>> entry,
274       @NonNull YAMLGenerator generator) throws IOException {
275     IDefinitionNodeItem<?, ?> target = ObjectUtils.notNull(entry.getKey());
276 
277     List<AllowedValuesRecord> allowedValues = entry.getValue();
278     if (allowedValues != null) {
279       generator.writeFieldName("constraints");
280 
281       generator.writeStartArray(); // constraints
282 
283       for (AllowedValuesRecord record : allowedValues) {
284         assert target.equals(record.getTarget());
285 
286         writeAllowedValue(record, generator);
287       }
288 
289       generator.writeEndArray(); // constraints
290     }
291   }
292 
293   private static void writeAllowedValue(@NonNull AllowedValuesRecord record, @NonNull YAMLGenerator generator)
294       throws IOException {
295 
296     generator.writeStartObject(); // constraint
297 
298     generator.writeStringField("type", "allowed-values");
299 
300     IAllowedValuesConstraint constraint = record.getAllowedValues();
301     if (constraint.getId() != null) {
302       generator.writeStringField("identifier", constraint.getId());
303     }
304     generator.writeStringField("location", metapath(record.getLocation()));
305     generator.writeStringField("target", constraint.getTarget());
306 
307     List<String> values = constraint.getAllowedValues().values().stream()
308         .map(IAllowedValue::getValue)
309         .collect(Collectors.toList());
310     generator.writeFieldName("values");
311     if (values == null) {
312       generator.writeNull();
313     } else {
314       generator.writeStartArray();
315       for (String value : values) {
316         generator.writeString(value);
317       }
318       generator.writeEndArray();
319     }
320 
321     generator.writeBooleanField("allow-other", constraint.isAllowedOther());
322 
323     URI source = constraint.getSource().getSource();
324     generator.writeStringField("source", source == null ? "builtin" : source.toString());
325 
326     generator.writeEndObject(); // constraint
327   }

329   private static String metapath(@NonNull IDefinitionNodeItem<?, ?> item) {
330     return metapath(item.getMetapath());
331   }
332 
333   private static String metapath(@NonNull String path) {
334     // remove position 1 predicates
335     return path.replace("[1]", "");
336   }
