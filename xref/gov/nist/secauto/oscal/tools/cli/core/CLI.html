29  import java.util.LinkedHashMap;
30  import java.util.Map;
31  
32  import edu.umd.cs.findbugs.annotations.NonNull;
33  
34  @SuppressWarnings("PMD.ShortClassName")
35  public final class CLI {
36    public static void main(String[] args) {
37      System.setProperty("java.util.logging.manager", "org.apache.logging.log4j.jul.LogManager");
38  
39      ExitStatus status = runCli(args);
40      int exitCode = status.getExitCode().getStatusCode();
41      System.exit(exitCode);
42    }
43  
44    @NonNull
45    public static ExitStatus runCli(String... args) {
46      @SuppressWarnings("PMD.UseConcurrentHashMap")
47      Map<String, IVersionInfo> versions = new LinkedHashMap<>();
48      versions.put(CLIProcessor.COMMAND_VERSION, new OscalCliVersion());
49      versions.put("https://github.com/usnistgov/liboscal-java", new LibOscalVersion());
50      versions.put("https://github.com/usnistgov/OSCAL", new OscalVersion());
51      versions.put("https://github.com/usnistgov/metaschema-java", new MetaschemaJavaVersion());
52      versions.put("https://github.com/usnistgov/metaschema", new MetaschemaVersion());
53  
54      CLIProcessor processor = new CLIProcessor("oscal-cli", CollectionUtil.unmodifiableMap(versions));
55      processor.addCommandHandler(new CatalogCommand());
56      processor.addCommandHandler(new ProfileCommand());
57      processor.addCommandHandler(new ComponentDefinitionCommand());
58      processor.addCommandHandler(new SystemSecurityPlanCommand());
59      processor.addCommandHandler(new AssessmentPlanCommand());
60      processor.addCommandHandler(new AssessmentResultsCommand());
61      processor.addCommandHandler(new PlanOfActionsAndMilestonesCommand());
62      processor.addCommandHandler(new MetaschemaCommand());
63      processor.addCommandHandler(new ValidateCommand());
64      processor.addCommandHandler(new ConvertCommand());
65      processor.addCommandHandler(new ResolveCommand());
66      processor.addCommandHandler(new ListAllowedValuesCommand());
67      return processor.process(args);
68    }
69  
70    private CLI() {
71      // disable construction
72    }
73  }
